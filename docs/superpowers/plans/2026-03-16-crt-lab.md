# crt-lab Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `crt-lab`, an interactive bash TUI for live-tuning 13 CRT shader parameters in Ghostty, with 7 built-in presets and animated scanlines via `iTime`.

**Architecture:** Single bash script (`scripts/crt-lab`) embeds the GLSL body as a heredoc and generates `~/.config/ghostty/shaders/crt-lab.glsl` by prepending a `#define` block on each `r` keypress. The TUI renders a parameter list with bar graphs; arrow keys select/nudge values; `n`/`p` switch presets. Config-patching and cleanup follow the existing `crt-cycle` pattern exactly.

**Tech Stack:** bash 5 (zsh-compatible), `bc` for float math, `tput`/ANSI for TUI, `osascript` + `screencapture` for macOS integration, GLSL 1.20 (Ghostty/Shadertoy uniform set: `iTime`, `iResolution`, `iChannel0`).

**Spec:** `docs/superpowers/specs/2026-03-16-crt-lab-design.md`

// crt-lab.glsl — animated CRT shader for Ghostty
// Generated header (defines) prepended by crt-lab script.
// Uniforms provided by Ghostty: iTime (float), iResolution (vec2), iChannel0 (sampler2D)
//
// REQUIRES these defines (prepended by script):
//   ROWS, COLS, CURVATURE, SCAN_INTENSITY, SCAN_DENSITY, SCAN_SPEED,
//   PHOSPHOR_DECAY, FLICKER_AMP, CHROMA_SHIFT, BLOOM, BLOOM_RADIUS,
//   DOT_MATRIX, VIGNETTE, BRIGHTNESS, GRAIN_AMP
//   (optional) PHOSPHOR_GREEN

#define PI 3.14159265359

void mainImage(out vec4 fragColor, in vec2 fragCoord)
{
    vec2 uv = fragCoord / iResolution.xy;

    // --- Internal parameter remapping ---
    float curvature = CURVATURE * CURVATURE * 0.08;
    float scanStr   = SCAN_INTENSITY * 0.7;
    float chromaOff = CHROMA_SHIFT * CHROMA_SHIFT * 0.005;
    float bloomStr  = BLOOM;
    float bloomRad  = mix(4.0, 24.0, BLOOM_RADIUS);
    float dotStr    = DOT_MATRIX * 0.5;
    float vigPow    = mix(0.001, 0.35, VIGNETTE * VIGNETTE);
    float bright    = mix(0.85, 1.35, BRIGHTNESS);
    float grainAmp  = GRAIN_AMP * 0.04;
    float flickAmp  = FLICKER_AMP * 0.04;

    // --- Curvature ---
    vec2 curved = uv * 2.0 - 1.0;
    curved *= 1.0 + dot(curved, curved) * curvature;
    vec2 cuv = curved * 0.5 + 0.5;
    if (cuv.x < 0.0 || cuv.x > 1.0 || cuv.y < 0.0 || cuv.y > 1.0) {
        fragColor = vec4(texture(iChannel0, vec2(0.0)).rgb, 1.0);
        return;
    }

    // --- Chromatic Aberration ---
    vec3 color;
    if (chromaOff > 0.0) {
        color = vec3(
            texture(iChannel0, cuv + vec2( chromaOff, 0.0)).r,
            texture(iChannel0, cuv).g,
            texture(iChannel0, cuv + vec2(-chromaOff, 0.0)).b
        );
    } else {
        color = texture(iChannel0, cuv).rgb;
    }

    // --- Bloom: additive glow from bright neighbours ---
    if (bloomStr > 0.0) {
        vec3 bloom = vec3(0.0);
        float total = 0.0;
        for (float x = -3.0; x <= 3.0; x += 1.0) {
            for (float y = -3.0; y <= 3.0; y += 1.0) {
                float w = exp(-0.5 * (x * x + y * y));
                bloom += texture(iChannel0, cuv + vec2(x, y) / iResolution.xy * (bloomRad / 3.0)).rgb * w;
                total += w;
            }
        }
        bloom /= total;
        color += max(bloom - color, vec3(0.0)) * bloomStr;
    }

    // --- Animated Scanlines (phosphor beam simulation) ---
    // scrollRate: sqrt remap so perceptual change is linear across 0-1 slider
    float scanCompensation = 1.0;
    if (scanStr > 0.0) {
        float scrollRate = sqrt(SCAN_SPEED) * 150.0;
        float scrolledY  = fragCoord.y - iTime * scrollRate;
        float period     = floor(mix(2.0, 8.0, 1.0 - SCAN_DENSITY));
        float phase      = mod(scrolledY, period) / period;  // 0..1 per scan period

        // sineScan: symmetric reference profile (PHOSPHOR_DECAY=0)
        float sineScan = pow(sin(phase * PI), 2.0);

        // decayScan: asymmetric phosphor decay (beam scans downward, phosphor fades after)
        // decayExp controls falloff steepness; normalized so peak ≈ 1.0
        float decayExp  = sqrt(PHOSPHOR_DECAY) * 8.0;
        float decayNorm = 1.0 / max(1.0 - exp(-decayExp), 0.001);  // stability guard
        float decayScan = exp(-phase * decayExp) * (1.0 - exp(-decayExp)) * decayNorm;

        // mix(sineScan, decayScan, 0) = sineScan — no div-by-zero risk at PHOSPHOR_DECAY=0
        float beamProfile = mix(sineScan, decayScan, PHOSPHOR_DECAY);
        float scanline    = 1.0 - scanStr * beamProfile;
        color *= scanline;
        scanCompensation  = 1.0 / max(1.0 - scanStr * 0.5, 0.001);
    }

    // --- Flicker: 60Hz brightness oscillation ---
    // At max FLICKER_AMP=1, amplitude = 4% brightness swing — below strobing threshold
    if (flickAmp > 0.0) {
        color *= 1.0 + flickAmp * sin(iTime * 2.0 * PI * 60.0);
    }

    // --- Dot Matrix: one RGB cycle per character cell ---
    float dotCompensation = 1.0;
    if (dotStr > 0.0) {
        float lineHeight = iResolution.y / float(ROWS);
        float charWidth  = iResolution.x / float(COLS);
        float dx = sin(fragCoord.x * PI / charWidth);
        float dy = sin(fragCoord.y * PI / lineHeight);
        float dotPattern = 1.0 - dotStr * (0.5 + 0.5 * dx * dy);
        color *= dotPattern;
        dotCompensation = 1.0 / max(1.0 - dotStr * 0.25, 0.001);
    }

    // --- Vignette ---
    if (vigPow > 0.0) {
        vec2 vig = cuv * (1.0 - cuv);
        color *= clamp(pow(vig.x * vig.y * 15.0, vigPow), 0.0, 1.0);
    }

    // --- Grain ---
    if (grainAmp > 0.0) {
        float gr = fract(sin(dot(fragCoord.xy + iTime * 743.0,
                                 vec2(12.9898, 78.233))) * 43758.5453) - 0.5;
        color += gr * grainAmp;
    }

    // --- Phosphor Green Tint (P31 phosphor simulation) ---
#ifdef PHOSPHOR_GREEN
    float luma = dot(color, vec3(0.2126, 0.7152, 0.0722));
    color = luma * vec3(0.20, 1.0, 0.28);
#endif

    // --- Brightness with auto-compensation for scan/dot darkening ---
    color *= scanCompensation * dotCompensation * bright;

    fragColor = vec4(clamp(color, 0.0, 1.0), 1.0);
}
GLSL

#!/usr/bin/env bash
#
# crt-lab — interactive CRT shader parameter tuner for Ghostty
#
# Usage:
#   crt-lab            Start on preset 1 (Clean)
#   crt-lab --preset N Start on preset N (1-indexed)
#
# Keys: ↑↓ select param  ←→ nudge  [/] coarse  n/p preset  r reload  s screenshot  w save  q quit
#
set -uo pipefail

# ── Paths ─────────────────────────────────────────────────────────────────────
SHADERS_DIR="$HOME/.config/ghostty/shaders"
SHADER_OUT="$SHADERS_DIR/crt-lab.glsl"
SHADER_BODY="$SHADERS_DIR/crt-lab.glsl"  # same file; script owns the full output
REAL_CONFIG="$HOME/Library/Application Support/com.mitchellh.ghostty/config"
DOTS_CONFIG="$HOME/Development/dotfiles/configs/ghostty/config"
SESSION_DIR="/tmp/crt-lab-$$"
SHOT_COUNTER=0
SHOT_STATUS=""

# ── Config patching ───────────────────────────────────────────────────────────

patch_configs() {
    local from="$1" to="$2"
    sed -i '' "s|${from}|${to}|g" "$REAL_CONFIG" 2>/dev/null || true
    sed -i '' "s|${from}|${to}|g" "$DOTS_CONFIG" 2>/dev/null || true
}

# ── Cleanup ───────────────────────────────────────────────────────────────────

cleanup() {
    tput cnorm 2>/dev/null
    printf "\n\033[0m"
    patch_configs 'crt-lab\.glsl' 'crt-clean.glsl'
    # Print screenshots if any were taken
    if [[ -d "$SESSION_DIR" ]] && compgen -G "$SESSION_DIR/*.png" > /dev/null 2>&1; then
        echo ""
        echo "Screenshots saved:"
        for f in "$SESSION_DIR"/*.png; do
            echo "  $f"
        done
    fi
}
trap cleanup EXIT INT TERM

SCRIPT

# ── TUI rendering ─────────────────────────────────────────────────────────────

# Render a 16-char progress bar for value 0.0..1.0
render_bar() {
    local val=$1
    # Convert to integer 0..16 using bc (avoids subshell rounding issues)
    local filled
    filled=$(echo "scale=0; ($val * 16 + 0.5) / 1" | bc 2>/dev/null)
    filled=${filled:-0}
    [[ $filled -lt 0 ]] && filled=0
    [[ $filled -gt 16 ]] && filled=16
    local empty=$((16 - filled))
    local bar=""
    local i
    for ((i=0; i<filled; i++)); do bar+="█"; done
    for ((i=0; i<empty; i++)); do bar+="░"; done
    printf '%s' "$bar"
}

show_ui() {
    clear
    local pgreen_tag=""
    [[ "${PRESET_PGREEN[$PRESET_IDX]}" == "1" ]] && pgreen_tag=" \033[32m[P31]\033[0m"

    # Header
    printf "\033[1;36m CRT Lab \033[0m\033[2m─────────────────────────\033[0m"
    printf " \033[1;33m%s\033[0m%b \033[2m[%d/%d]\033[0m\n" \
        "${PRESET_NAMES[$PRESET_IDX]}" "$pgreen_tag" $((PRESET_IDX+1)) $TOTAL_PRESETS

    # Key hint
    printf "\n \033[2mn/p preset   ↑↓ select   ←→ nudge   [/] coarse\033[0m\n"
    printf " \033[2mr reload   s screenshot   w save   q quit\033[0m\n\n"

    # Parameter rows
    local i
    for ((i=0; i<${#PARAM_NAMES[@]}; i++)); do
        local cursor="  "
        [[ $i -eq $SELECTED ]] && cursor="\033[1;32m▸\033[0m "

        local anim=""
        [[ "${PARAM_ANIMATED[$i]}" == "1" ]] && anim="\033[36m⟳\033[0m"

        local bar
        bar=$(render_bar "${PARAM_VALS[$i]}")

        printf " %b\033[%-15s\033[0m \033[33m%s\033[0m  \033[2m%4s\033[0m  %b  \033[2m%s\033[0m\n" \
            "$cursor" "${PARAM_NAMES[$i]}" "$bar" "${PARAM_VALS[$i]}" "$anim" "${PARAM_DESCS[$i]}"
    done

    # Shot status
    if [[ -n "$SHOT_STATUS" ]]; then
        printf "\n %b\n" "$SHOT_STATUS"
    fi

    printf "\n "
}

SCRIPT

# ── Input ─────────────────────────────────────────────────────────────────────

# Reads one keypress. Sets global KEY to one of:
#   UP DOWN LEFT RIGHT LBRACKET RBRACKET or the literal character
read_key() {
    local k b1 b2
    IFS= read -r -s -n 1 k
    if [[ "$k" == $'\x1b' ]]; then
        # Read up to 2 more bytes with timeout (handles lone ESC gracefully)
        IFS= read -r -s -n 1 -t 0.15 b1 || true
        if [[ "$b1" == "[" ]]; then
            IFS= read -r -s -n 1 -t 0.15 b2 || true
            case "$b2" in
                A) KEY="UP"    ;;
                B) KEY="DOWN"  ;;
                C) KEY="RIGHT" ;;
                D) KEY="LEFT"  ;;
                *) KEY="ESC"   ;;
            esac
        else
            KEY="ESC"
        fi
    else
        KEY="$k"
    fi
}

    case "$1" in
        --preset) START_PRESET=$(( $2 - 1 )); shift 2 ;;
        -h|--help)
            echo "Usage: crt-lab [--preset N]"
            echo "  ↑↓ select param  ←→ nudge  [/] coarse nudge"
            echo "  n/p preset  r reload Ghostty  s screenshot  w save  q quit"

    exit 1

    read_key

    case "$KEY" in

            ;;
        DOWN)
            SELECTED=$(( (SELECTED + 1) % NPARAMS ))
            ;;
        LEFT)
            nudge_param "$SELECTED" "-${PARAM_FINE[$SELECTED]}"
            write_shader
            ;;
        RIGHT)
            nudge_param "$SELECTED" "${PARAM_FINE[$SELECTED]}"
            write_shader
            ;;
        "[")
            nudge_param "$SELECTED" "-${PARAM_COARSE[$SELECTED]}"
            write_shader
            ;;
        "]")
            nudge_param "$SELECTED" "${PARAM_COARSE[$SELECTED]}"
            write_shader
            ;;
        n|N)
            PRESET_IDX=$(( (PRESET_IDX + 1) % TOTAL_PRESETS ))
            load_preset "$PRESET_IDX"
            write_shader
            ;;
        p|P)
            PRESET_IDX=$(( (PRESET_IDX - 1 + TOTAL_PRESETS) % TOTAL_PRESETS ))
            load_preset "$PRESET_IDX"
            write_shader
            ;;
        r|R)
            write_shader
            reload_ghostty
            ;;
        s|S)
            tput cnorm
            take_screenshot
            tput civis
            ;;
        w|W)
            write_shader
            printf "\033[32m✓ Saved\033[0m to crt-lab.glsl\n\n"
            # Print equivalent crt-tune command for the 7 shared params
            local warp_flag="--warp"
            printf "crt-tune --scale 0.33 --thin %.2f --blur %.2f --mask %.2f --min-vin %.2f %s --mask-type shadow\n\n" \
                "${PARAM_VALS[1]}" "-2.75" "${PARAM_VALS[7]}" "${PARAM_VALS[10]}" "$warp_flag"
            sleep 2
            ;;
        q|Q)
            exit 0
            ;;
    esac
done

take_screenshot() {
    local safe_name
    safe_name=$(printf '%s' "${PRESET_NAMES[$PRESET_IDX]}" | tr -cs '[:alnum:]' '_')
    SHOT_COUNTER=$(( SHOT_COUNTER + 1 ))
    local outfile
    outfile=$(printf '%s/%02d_%s.png' "$SESSION_DIR" "$SHOT_COUNTER" "$safe_name")

    mkdir -p "$SESSION_DIR"

    # Get Ghostty window bounds via AppleScript
    local pos siz
    pos=$(osascript -e 'tell application "System Events" to tell process "Ghostty" to get position of front window' 2>/dev/null)
    siz=$(osascript -e 'tell application "System Events" to tell process "Ghostty" to get size of front window' 2>/dev/null)
    if [[ -z "$pos" || -z "$siz" ]]; then
        SHOT_STATUS=$'\033[31m✗ Could not get Ghostty window bounds (grant Accessibility in System Settings)\033[0m'
        return 1
    fi

    local x1 y1 w h
    read -r x1 y1 <<< "$(echo "$pos" | tr -cs '0-9\n' ' ')"
    read -r w  h  <<< "$(echo "$siz" | tr -cs '0-9\n' ' ')"

    local err
    err=$(screencapture -x -R "${x1},${y1},${w},${h}" "$outfile" 2>&1) || {
        SHOT_STATUS=$'\033[31m✗ screencapture failed'"${err:+ — $err}"$'\033[0m'
        return 1
    }

    # Detect empty file (permission failure)
    local sz
    sz=$(wc -c < "$outfile" 2>/dev/null || echo 0)
    if [[ $sz -lt 5000 ]]; then
        SHOT_STATUS=$'\033[31m✗ Screenshot empty — grant Screen Recording in System Settings → Privacy\033[0m'
        rm -f "$outfile"
        return 1
    fi

    # Resize to 1500px wide (keeps Retina CRT detail, manageable file size)
    sips --resampleWidth 1500 "$outfile" &>/dev/null || true

    SHOT_STATUS="\033[32m✓\033[0m $(basename "$outfile")"
}
